const float saturation 	= 1.1f;
const float contrast 	= 1.1f;
const float brightness 	= 1.0f;


float3 c = tex2D( samp0, tex0 ).rgb;

float3 gray = ( c.r + c.g + c.b ) / 3;
c = pow( c, brightness ) * contrast;
c = lerp( gray, c, saturation );

float4 d = tex2D( samp1, tex0 );
c = lerp( c.rgb, d.rgb, d.a );

return float4( c, 1.0f );
